// 6. Header simulation
function sendHeader() {
    let headerVal = document.getElementById("headerInput").value;
    fetch(window.location.href, {
        method: "GET",
        headers: { "X-WAF-Test": headerVal }
    });
    document.getElementById("headerOutput").innerText = "Header sent: " + headerVal;
}

// 7. Command injection simulation
function sendCmd() {
    let cmd = encodeURIComponent(document.getElementById("cmdInput").value);
    fetch("/?cmd=" + cmd);
    document.getElementById("cmdOutput").innerText = "Command sent: " + cmd;
}

// 8. Path traversal
function sendPath() {
    let path = encodeURIComponent(document.getElementById("pathInput").value);
    fetch("/?path=" + path);
    document.getElementById("pathOutput").innerText = "Path sent: " + path;
}

// 9. Cookie injection
function setCookie() {
    let val = document.getElementById("cookieInput").value;
    document.cookie = val;
    document.getElementById("cookieOutput").innerText = "Cookie set: " + val;
}

// 10. Redirect
function redirect() {
    let url = document.getElementById("redirectInput").value;
    window.location.href = url;
